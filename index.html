package main

import (
        "fmt"
        "math/rand"
)

// ListNode represents a node in a singly-linked list.
type ListNode struct {
        Val  int
        Next *ListNode
}

// generateLinkedList creates a linked list of length `n` with random values (0-9).
func generateLinkedList(n int) *ListNode {
        if n <= 0 {
                return nil
        }

        head := &ListNode{Val: rand.Intn(10)} // Random digit (0-9)
        current := head

        for i := 1; i < n; i++ {
                current.Next = &ListNode{Val: rand.Intn(10)}
                current = current.Next
        }

        return head
}

// printLinkedList prints the linked list values.
func printLinkedList(head *ListNode) {
        current := head
        for current != nil {
                fmt.Print(current.Val)
                if current.Next != nil {
                        fmt.Print(" -> ")
                }
                current = current.Next
        }
        fmt.Println()
}

func main() {
        // Generate a linked list of length 5.
        list := generateLinkedList(5)
        list2 := generateLinkedList(5)
        printLinkedList(list)  // Example output: 3 -> 7 -> 1 -> 4 -> 2
        printLinkedList(list2) // Example output: 3 -> 7 -> 1 -> 4 -> 2
        list3 := sumLists(list, list2)
        printLinkedList(list3) // Example output: 3 -> 7 -> 1 -> 4 -> 2

}

func sumLists(l1, l2 *ListNode) *ListNode {

        l3h := &ListNode{}
        l3 := l3h

        var a, b, c int
        for l1 != nil || l2 != nil {
                l3.Next = &ListNode{}
                l3 = l3.Next
                if l1 != nil {
                        a = l1.Val
                        l1 = l1.Next
                }

                if l2 != nil {
                        b = l2.Val
                        l2 = l2.Next
                }
                c = a + b
                l3.Val = c

                a, b = 0, 0

        }
        return l3h.Next
}
